
### 方式2， 直接配置数据源与缓存

server.port: 7281

  name: gritapi
  group: grit

  driverType: "local"

  schema: water
  url: "jdbc:mysql://localhost:3306/water?useSSL=false&allowMultiQueries=true&useUnicode=true&characterEncoding=utf8&autoReconnect=true&rewriteBatchedStatements=true"
  password: "1234"
  username: "demo"

grit.token: B6uWZDYUm4kMscdEAERXQ2wMBW1nLL0j

solon.logging.logger:
  "com.zaxxer.hikari.*":
