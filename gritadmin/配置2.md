
### 方式2， 直接配置数据源与缓存

```yaml
server.port: 7282

solon.app:
  name: gritadmin
  group: grit
  title: "GRIT 权限管理"

grit.cache:
  driverType: "local"

grit.db:
  schema: water
  url: "jdbc:mysql://localhost:3306/water?useSSL=false&allowMultiQueries=true&useUnicode=true&characterEncoding=utf8&autoReconnect=true&rewriteBatchedStatements=true"
  password: "1234"
  username: "demo"

solon.logging.logger:
  "com.zaxxer.hikari.*":
    level: "WARN"

gritadmin:
  user: admin
  password: SykSYLWN9WTpzCHq
```
